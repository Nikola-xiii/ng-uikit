LoopMe Angular 2 UIKit

Install

    
    npm install loopme/uikit

Include and use

Import basic layout styles

In main styles.scss file

      
        @import "../components/layouts/layouts.component";

Import NgModule

    
    import { NgUIKitModule } from 'ng-uikit/components/ng-uikit.module';
    
  
    
      import { NgUIKitModule } from 'ng-uikit/components/ng-uikit.module';

      @NgModule({
        ...
        imports: [
        ...,
        NgUIKitModule,
      ]});
